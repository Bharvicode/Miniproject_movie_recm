  <script>
    const form = document.getElementById("movieForm");
    const result = document.getElementById("result");

    const movieList = {
      action: ["Marvel", "Pathan", "Animal", "URI", "Avengers: Endgame", "Mission Impossible", "Black Panther", "Top Gun: Maverick"],
      comedy: ["Phir Hera Pheri", "Golmaal", "PK", "Mr. Bean's Holiday", "21 Jump Street", "3 Idiots", "Ted"],
      drama: ["Yashoda", "Hum Aapke Hai Kaun", "Vivah", "The Green Mile", "Titanic", "The Pursuit of Happyness", "The Social Network"],
      horror: ["The Conjuring", "Shaitan", "Vash", "It", "1920", "Annabelle", "Sinister", "Hereditary"],
      scifi: ["Mission Mangal", "Avatar", "The Matrix", "Robot", "Dune", "Edge of Tomorrow", "Blade Runner 2049", "Star Wars"]
    };

    function validateForm() {
      let isValid = true;
      const name = document.getElementById("name").value.trim();
      const age = document.getElementById("age").value.trim();
      const genre = document.getElementById("genre").value;
      const nameError = document.getElementById("nameError");
      const ageError = document.getElementById("ageError");
      const genreError = document.getElementById("genreError");

      nameError.textContent = "";
      ageError.textContent = "";
      genreError.textContent = "";

      if (name === "") {
        nameError.textContent = "Name is required.";
        isValid = false;
      } else if (name.length < 3) {
        nameError.textContent = "Name must be at least 3 characters long.";
        isValid = false;
      }

      if (age === "") {
        ageError.textContent = "Age is required.";
        isValid = false;
      } else if (isNaN(age) || age < 13 || age > 120) {
        ageError.textContent = "Please enter a valid age between 13 and 120.";
        isValid = false;
      }

      if (genre === "") {
        genreError.textContent = "Please select a genre.";
        isValid = false;
      }

      return isValid;
    }

    function getRandomMovies(list, count) {
      const shuffled = [...list].sort(() => 0.5 - Math.random());
      return shuffled.slice(0, count);
    }

    form.addEventListener("submit", function (e) {
      e.preventDefault();

      if (!validateForm()) {
        result.innerHTML = "";
        return;
      }

      const name = document.getElementById("name").value.trim();
      const age = document.getElementById("age").value.trim();
      const genre = document.getElementById("genre").value;
      const movies = movieList[genre];
      const randomMovies = getRandomMovies(movies, 3);

      result.innerHTML = `
        <h3>Hello, ${name}!</h3>
        <p>Age: <b>${age}</b></p>
        <p>You love <b>${genre.toUpperCase()}</b> movies ðŸŽ¬</p>
        <p>Here are some recommendations for you:</p>
        <ul>${randomMovies.map(movie => <li>âœ¨ ${movie}</li>).join('')}</ul>
      `;

      form.reset();
    });
  </script>
